<script>
const chatBody = document.getElementById('chatBody');
const data = { name: '', phone: '', issue: '' };

function scrollDown() {
  chatBody.scrollTop = chatBody.scrollHeight;
}

function addBotMessage(text, delay = 700) {
  return new Promise(resolve => {
    const typing = document.createElement('div');
    typing.className = 'typing';
    typing.innerHTML = '<span></span><span></span><span></span>';
    chatBody.appendChild(typing);
    scrollDown();

    setTimeout(() => {
      typing.remove();
      const msg = document.createElement('div');
      msg.className = 'msg bot';
      msg.textContent = text;
      chatBody.appendChild(msg);
      scrollDown();
      resolve();
    }, delay);
  });
}

function addUserMessage(text) {
  const msg = document.createElement('div');
  msg.className = 'msg user';
  msg.textContent = text;
  chatBody.appendChild(msg);
  scrollDown();
}

function addOptions(options, onSelect) {
  return new Promise(resolve => {
    const wrap = document.createElement('div');
    wrap.className = 'options';
    options.forEach(opt => {
      const btn = document.createElement('button');
      btn.className = 'option-btn';
      btn.textContent = opt;
      btn.onclick = () => {
        Array.from(wrap.children).forEach(b => b.disabled = true);
        addUserMessage(opt);
        wrap.remove();
        onSelect(opt);
        resolve(opt);
      };
      wrap.appendChild(btn);
    });
    chatBody.appendChild(wrap);
    scrollDown();
  });
}

function addTextInput(placeholder, onSubmit) {
  return new Promise(resolve => {
    const wrap = document.createElement('div');
    wrap.className = 'input-row';
    wrap.innerHTML = `
      <input type="text" class="text-input" placeholder="${placeholder}" />
      <button class="send-btn">Send</button>
    `;
    const input = wrap.querySelector('input');
    const btn = wrap.querySelector('button');

    function submit() {
      const val = input.value.trim();
      if (!val) return;
      addUserMessage(val);
      wrap.remove();
      onSubmit(val);
      resolve(val);
    }

    btn.onclick = submit;
    input.onkeydown = e => { if (e.key === 'Enter') submit(); };
    chatBody.appendChild(wrap);
    scrollDown();
    input.focus();
  });
}

async function runFlow() {
  await addBotMessage("Hi! Thanks for reaching out to HVAC Company — sorry we missed your call. What can we help you with today?");

  const service = await addOptions(
    ["AC Not Cooling", "Heater Not Working", "New Installation", "Something Else"],
    () => {}
  );

  if (service === "AC Not Cooling") {
    data.issue = "AC not cooling";
    await addBotMessage("Thank you. We'll prioritize getting your cooling system back up and running.");
  } else if (service === "Heater Not Working") {
    data.issue = "Heater not working";
    await addBotMessage("Thank you. We understand heating issues need quick attention, and we'll get this addressed.");
  } else if (service === "New Installation") {
    data.issue = "New installation";
    await addBotMessage("Thank you. We'd be happy to assist with a new installation. A few quick details first.");
  } else {
    await addBotMessage("Thank you. Could you briefly describe what you need help with?");
    const desc = await addTextInput("Describe your issue...", () => {});
    data.issue = desc;
    await addBotMessage("Thank you — our technicians handle a wide range of heating and cooling issues, so we'll make sure this gets addressed.");
  }

  const urgency = await addOptions(["Urgent", "Can Wait"], () => {});
  data.urgency = urgency;

  await addBotMessage("Understood. Could I get your name, please?");
  const name = await addTextInput("Your name...", () => {});
  data.name = name;

  await addBotMessage(`Thank you, ${name}. And the best phone number to reach you?`);
  const phone = await addTextInput("Your phone number...", () => {});
  data.phone = phone;

  await addBotMessage(`Thank you, ${name}. Your request has been received, and our team will contact you shortly.`, 900);

  console.log("Captured lead:", data);
}

runFlow();
